cmake_minimum_required(VERSION 3.10)
project(OCCTPointOnLine)

# 设置C++标准
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 查找OCCT包
find_package(OpenCASCADE REQUIRED)

# 添加可执行文件
add_executable(OCCTPointOnLine OCCTPointOnLine.cpp)

# 链接OCCT库
target_link_libraries(OCCTPointOnLine OpenCASCADE::OpenCASCADE)

# 设置编译选项
target_compile_options(OCCTPointOnLine PRIVATE -Wall -Wextra)

# 设置输出目录
set_target_properties(OCCTPointOnLine PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin
)

# 安装规则
install(TARGETS OCCTPointOnLine
    RUNTIME DESTINATION bin
)
