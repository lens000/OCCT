cmake_minimum_required(VERSION 3.10)
project(OCCTPointOnLine)

# 设置C++标准
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 设置OCCT根目录
set(OCCT_ROOT_DIR ${CMAKE_CURRENT_SOURCE_DIR}/..)

# 包含OCCT的CMake模块
set(CMAKE_MODULE_PATH "${OCCT_ROOT_DIR}/adm/cmake")

# 包含OCCT工具包
include(occt_toolkit)

# 添加可执行文件
add_executable(OCCTPointOnLine OCCTSimple.cpp)

# 链接OCCT库
target_link_libraries(OCCTPointOnLine TKernel TKMath)

# 设置编译选项
target_compile_options(OCCTPointOnLine PRIVATE -Wall -Wextra)

# 设置输出目录
set_target_properties(OCCTPointOnLine PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin
)
